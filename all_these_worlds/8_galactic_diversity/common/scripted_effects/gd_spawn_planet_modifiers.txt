spawn_gravity_modifier = {
	# Spawn gravity modifier
	if = {
		limit = { has_modifier = low_gravity }
		remove_modifier = low_gravity #in case the vanilla modifier spawned already
	}
	if = {
		limit = { has_modifier = high_gravity }
		remove_modifier = high_gravity #in case the vanilla modifier spawned already
	}
	if = {
		limit = {
			OR = {
				# Add special planets we dont want random gravities here
				has_planet_flag = "planet_venus"
				has_planet_flag = "planet_mars"
				has_planet_flag = "planet_earth"
				has_planet_flag = "planet_unity"
				has_planet_flag = "moon_luna"
				has_planet_flag = "moon_europa"
				has_planet_flag = "moon_titan"
				#is_planet_class = "pc_megachloridal"
				#is_planet_class = "orbit_megachloridal"
			}
		}
		random_list = {
			0 = {
				add_modifier = {
					modifier = "low_gravity"
					days = -1
				}
				modifier = { add = 100 has_planet_flag = "planet_mars" }
				modifier = { add = 100 has_planet_flag = "moon_luna" }
				modifier = { add = 100 has_planet_flag = "moon_europa" }
				modifier = { add = 100 has_planet_flag = "moon_titan" }
				#modifier = { add = 100 is_planet_class = "pc_megachloridal" }
				#modifier = { add = 100 is_planet_class = "orbit_megachloridal" }
			}
			0 = {
				add_modifier = {
					modifier = "standard_gravity"
					days = -1
				}
				modifier = { add = 100 has_planet_flag = "planet_venus" }
				modifier = { add = 100 has_planet_flag = "planet_earth" }
				modifier = { add = 100 has_planet_flag = "planet_unity" }
			}
		}
		set_planet_flag = "gravity_spawned"
	}
	else = {
		random_list = {
			0 = {
				add_modifier = {
					modifier = "low_gravity"
					days = -1
				}
				modifier = { add = 20 planet_size < 16 }
				modifier = { add = 80 planet_size < 14 }
			}
			20 = {
				add_modifier = {
					modifier = "standard_gravity"
					days = -1
				}
				#modifier = { factor = 2.00 AND = { planet_size > 14 planet_size < 24 } }
			}
			0 = {
				add_modifier = {
					modifier = "high_gravity"
					days = -1
				}
				modifier = { add = 10 planet_size > 22 }
				modifier = { add = 40 planet_size > 24 }
			}
		}
		set_planet_flag = "gravity_spawned"
	}	
}

respawn_gravity_modifier = {
	# For those cases we just want to re-order the modifiers, sending the gravity to last
	if = {
		limit = { has_modifier = low_gravity }
		remove_modifier = low_gravity
		add_modifier = {
			modifier = "low_gravity"
			days = -1
		}
	}
	if = {
		limit = { has_modifier = standard_gravity }
		remove_modifier = standard_gravity
		add_modifier = {
			modifier = "standard_gravity"
			days = -1
		}
	}
	if = {
		limit = { has_modifier = high_gravity }
		remove_modifier = high_gravity
		add_modifier = {
			modifier = "high_gravity"
			days = -1
		}
	}	
}
