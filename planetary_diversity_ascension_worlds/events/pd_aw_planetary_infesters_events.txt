namespace = pdawplanetaryinfesters

# PD Planetary Infesters
# This event gives the Origin it's deposits, districts and modifiers at game start
country_event = {
	id = pdawplanetaryinfesters.10
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		is_country_type = default
		has_valid_civic = civic_hive_pd_planetary_infesters
	}
	immediate = {
		capital_scope = {
			change_pc = pc_hive
			validate_planet_buildings_and_districts = yes
			add_building = building_pd_aw_planet_infester_3
			while = {
				count = 2
				add_district = district_hive_1
			}
			add_zone = {
				district = district_hive_1
				zone = zone_energy
				zone_slot = 0
			}
			while = {
				count = 3
				add_district = district_hive_2
			}
			add_zone = {
				district = district_hive_2
				zone = zone_minerals
				zone_slot = 0
			}
			while = {
				count = 3
				add_district = district_hive_3
			}
			add_zone = {
				district = district_hive_3
				zone = zone_food
				zone_slot = 0
			}
		}
		every_galaxy_planet = {
			limit = {
				is_planet_class = pc_hive
				has_planet_flag = prescripted_ideal
			}
			change_pc = pc_arid
			reroll_planet = yes
		}
	}
}
