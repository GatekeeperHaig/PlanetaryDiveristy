building_seed_processing_center = {
	base_buildtime = @b1_time
	base_cap_amount = 1
	category = resource

	potential = {
		exists = owner
		owner = {
			OR = {
				country_uses_food = yes
				is_ai = no
			}
		}
		OR = {
			has_modifier = "biosphere-sparse"
			has_modifier = "biosphere-diverse"
			has_modifier = "biosphere-dense"
			has_modifier = "biosphere-exotic"
		}
	}
	
	allow = {
		exists = owner
		has_upgraded_capital = yes
	}

	destroy_trigger = {
		OR = {
			AND = {
				exists = owner
				OR = {
					owner = {
						is_ai = yes
						country_uses_food = no
					}
					AND = {
						owner = {
							is_ai = yes
						}
						has_any_farming_district_or_buildings = no
					}
				}
			}
			is_planet_class = pc_machine
			has_modifier = resort_colony
			buildings_no_motes = yes
		}
	}

	planet_modifier = {
		planet_farmers_seeds_produces_add = 1		
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = 2
		}
	}

}
