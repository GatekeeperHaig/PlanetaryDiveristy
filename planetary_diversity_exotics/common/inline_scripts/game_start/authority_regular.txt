# Individualist Empire Initial Setup
# Vanilla overridden to use Cryogenic Settlement Districts on pdexotic_cryo worlds.

if = {
	limit = {
		exists = owner
		owner = {
			is_regular_empire = yes
		}
	}
	add_building = building_capital

	if = {
		limit = { uses_district_set = pdexotic_cryo }
		add_district_and_planet_size_if_needed_effect_amount = {
			amount = 2
			district = district_pdexotic_cryo_city
		}
		add_zone = {
			district = district_pdexotic_cryo_city
			zone = zone_research_unity
			zone_slot = 1
		}
		add_zone = {
			district = district_pdexotic_cryo_city
			zone = zone_industrial
			zone_slot = 2
		}
		add_building = {
			building = building_factory_1
			zone = zone_industrial
		}
	}
	else = {
		add_district_and_planet_size_if_needed_effect_amount = {
			amount = 2
			district = district_city
		}
		add_zone = {
			district = district_city
			zone = zone_research_unity
			zone_slot = 1
		}
		add_zone = {
			district = district_city
			zone = zone_industrial
			zone_slot = 2
		}
		add_building = {
			building = building_factory_1
			zone = zone_industrial
		}
	}
}
