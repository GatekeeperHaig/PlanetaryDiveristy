		base = 2
	}

		category = planet_pops

					has_modifier = worse_parts
				}

			trade = 0.5
		}

			trigger = {
				exists = owner
				owner = { has_modifier = optimized_parts }
				is_sapient = yes
			}
			energy = 0.1
		}
	}

	custom_tooltip_with_modifiers = automodding_trait_cyborg_tooltip
