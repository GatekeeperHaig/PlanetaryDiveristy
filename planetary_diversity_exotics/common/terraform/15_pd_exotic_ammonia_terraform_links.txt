	from = "pc_pd_ammonia"

		category = terraforming
		inline_script = {
			script = terraform/triggered_costs

terraform_link = {
	from = "pc_pd_ammonia"
	to = "pc_machine"

	resources = {
		category = terraforming

		cost = {
			energy = @machineCost
		}
	}

	duration = @machineTime

	potential = {
		exists = owner
		has_ascension_perk = ap_machine_worlds
	}
	condition = {
		has_technology = "tech_pd_cryogenic_terraforming"
	}

	ai_weight = {
		weight = 10
		modifier = {
			factor = 0
			from = {
				any_owned_species = {
					NOR = {
						has_trait = trait_machine_unit
						has_trait = trait_mechanical
					}
					is_sapient = yes
				}
			}
		}
		modifier = {
			factor = 0
			owner = {
				any_owned_planet = { is_terraforming = yes }
			}
		}
		modifier = {
			factor = 0
			owner = {
				OR = {
					has_deficit = energy
					has_deficit = minerals
					has_deficit = food
					has_deficit = consumer_goods
				}
			}
		}
	}
}
