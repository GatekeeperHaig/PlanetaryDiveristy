# Converts an ammonia world to a pc_city arcology with the same mechanical outcome as the equivalent non-ammonia arcology decision: cryo urban districts roll over to district_arcology_housing on the planet-class swap, then get re-swapped to the garden / fortress / commercial housing district when that PD MA flavor modifier is present, then halved.
# Each flavor still uses its full PD MA modifier set and starter district / zone; the only cryo-specific touches are the ammonia-themed sky + planet entity selected by the portrait effect and the pd_ammonia_atmosphere modifier in the tooltip.
pd_exotic_cryo_arcology_convert_effect = {
	save_deposits = yes
	clear_deposits = yes
	restore_saved_deposits = yes
	change_pc = pc_city
	validate_planet_buildings_and_districts = yes
	if = {
		limit = { has_modifier = pd_garden_arcology }
		swap_districts_effect = {
			DISTRICT1 = district_arcology_housing
			DISTRICT2 = district_pd_garden_arcology_housing
		}
		relic_world_conversion_effect = {
			DISTRICT = district_pd_garden_arcology_housing
			VARIABLE = num_housing_districts
		}
	}
	else_if = {
		limit = { has_modifier = pd_fortress_arcology }
		swap_districts_effect = {
			DISTRICT1 = district_arcology_housing
			DISTRICT2 = district_pd_fortress_arcology_housing
		}
		relic_world_conversion_effect = {
			DISTRICT = district_pd_fortress_arcology_housing
			VARIABLE = num_housing_districts
		}
	}
	else_if = {
		limit = { has_modifier = pd_commercial_arcology }
		swap_districts_effect = {
			DISTRICT1 = district_arcology_housing
			DISTRICT2 = district_pd_commercial_arcology_housing
		}
		relic_world_conversion_effect = {
			DISTRICT = district_pd_commercial_arcology_housing
			VARIABLE = num_housing_districts
		}
	}
	else = {
		relic_world_conversion_effect = {
			DISTRICT = district_arcology_housing
			VARIABLE = num_housing_districts
		}
	}
	generate_pd_cryo_arcology_portrait_effect = yes
	if = {
		limit = {
			OR = {
				is_active_resolution = "resolution_ecology_environmental_control_board"
				is_active_resolution = "resolution_ecology_paradise_initiative"
			}
		}
		owner = {
			set_timed_country_flag = {
				flag = resolution_breached_terraformed
				days = 3600
			}
		}
	}
	validate_planet_buildings_and_districts = yes
}

# Overrides PD MA's remove_pd_capital_deposit_effect so demolishing a capital on an ammonia arcology returns to our cryo entity / sky instead of the standard PD MA city portrait. Non-ammonia planets fall through to the original PD MA portrait effects.
remove_pd_capital_deposit_effect = {
	if = {
		limit = {
			has_deposit = d_pd_senate_building
		}
		remove_deposit = d_pd_senate_building
		if = {
			limit = {
				OR = {
					pd_has_ammonia_atmosphere = yes
					pd_has_methane_atmosphere = yes
				}
			}
			generate_pd_cryo_arcology_portrait_effect = yes
		}
		else = {
			generate_pd_arcology_portrait_effect = yes
		}
	}
	if = {
		limit = {
			has_deposit = d_pd_palace_building
		}
		remove_deposit = d_pd_palace_building
		if = {
			limit = {
				OR = {
					pd_has_ammonia_atmosphere = yes
					pd_has_methane_atmosphere = yes
				}
			}
			generate_pd_cryo_arcology_portrait_effect = yes
		}
		else = {
			generate_pd_arcology_portrait_effect = yes
		}
	}
	if = {
		limit = {
			has_deposit = d_pd_headquarters_building
		}
		remove_deposit = d_pd_headquarters_building
		if = {
			limit = {
				OR = {
					pd_has_ammonia_atmosphere = yes
					pd_has_methane_atmosphere = yes
				}
			}
			generate_pd_cryo_arcology_portrait_effect = yes
		}
		else = {
			generate_pd_arcology_portrait_effect = yes
		}
	}
	if = {
		limit = {
			has_deposit = d_pd_confluence_spire_building
		}
		remove_deposit = d_pd_confluence_spire_building
		if = {
			limit = {
				OR = {
					pd_has_ammonia_atmosphere = yes
					pd_has_methane_atmosphere = yes
				}
			}
			generate_pd_cryo_arcology_portrait_effect = yes
		}
		else = {
			generate_pd_hive_arc_portrait_effect = yes
		}
	}
	if = {
		limit = {
			has_deposit = d_pd_machine_centralized_processing
		}
		remove_deposit = d_pd_machine_centralized_processing
		if = {
			limit = {
				OR = {
					pd_has_ammonia_atmosphere = yes
					pd_has_methane_atmosphere = yes
				}
			}
			generate_pd_cryo_arcology_portrait_effect = yes
		}
		else = {
			generate_pd_machine_arc_portrait_effect = yes
		}
	}
}
