# Planet-scope shortcuts for the cryogenic climate group and the broader exotic-world pool.
# Update these when new exotic climate groups (acidic / umbral / etc.) land.

is_pd_cryo_world = {
	OR = {
		is_planet_class = pc_pd_ammonia
		is_planet_class = pc_pd_methane
	}
}

is_pd_exotic_world = {
	OR = {
		is_planet_class = pc_pd_ammonia
		is_planet_class = pc_pd_methane
	}
}
