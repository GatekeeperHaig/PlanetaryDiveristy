# Planetary Diversity Exotics: Cryo chemist jobs
# Worker-tier farmer-class job tied to Chemovat Garden districts on cryogenic worlds. Produces food + exotic gases via cultivated chemosynthetic life.

pdexotic_cryo_chemist = {
	category = worker

	swappable_data = {
		default = {
			condition_string = WORKER_JOB_TRIGGER
			building_icon = building_chemical_plant
		}
	}

	tags = { food strategic_resource exotic_gases }

	possible_pre_triggers = {
		has_owner = yes
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible_precalc = can_fill_worker_job
	possible = {
		hidden_trigger = { exists = owner }
		owner = { is_gestalt = no }
	}

	resources = {
		category = planet_farmers
		produces = {
			food = 4
			exotic_gases = 0.3
		}
	}

	weight = {
		weight = @worker_job_weight
		mult = value:job_weights_modifier|JOB|pdexotic_cryo_chemist|RESOURCE|food|
		modifier = {
			factor = 0.2
			has_citizenship_rights = no
			NOT = { has_trait = trait_mechanical }
		}
	}
}

# Cryophile livestock variant. Mirrors livestock_lithoid / livestock_infernal: phenotype-gated job that returns the species' upkeep resource at a multiplier, plus the usual food yield. Covers the 0.05 gas/pop drain from trait_cryophile with a small margin so cryo-livestock is gas-positive for the captor.
livestock_cryophile = {
	category = worker
	first_come_first_served = no
	is_capped_by_modifier = no

	swappable_data = {
		default = {
			icon = livestock
			building_icon = building_food_processing_facility
		}
	}

	tags = { food strategic_resource exotic_gases }

	possible_pre_triggers = {
		is_enslaved = yes
		is_sapient = yes
		is_robotic = no
	}

	possible = {
		OR = {
			has_citizenship_type = {
				type = citizenship_slavery
				country = owner
			}
			has_citizenship_type = {
				type = citizenship_exploited_drone
				country = owner
			}
		}
		has_slavery_type = { type = slavery_livestock }
		has_trait = trait_cryophile
	}

	resources = {
		category = planet_livestock_farmers
		produces = {
			food = 4
			exotic_gases = 0.15
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_origin = origin_evolutionary_predators }
		}
		custom_tooltip = evolutionary_predators_pops_tt
	}

	weight = {
		weight = @special_job_weight
	}
}
