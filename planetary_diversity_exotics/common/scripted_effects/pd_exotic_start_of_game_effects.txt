# Adds starting deposits to a PD exotic homeworld. Invoked by pd_exotic.1 with the capital planet in scope.

generate_pd_exotic_start_deposits = {
	optimize_memory

	if = {
		limit = {
			is_planet_class = pc_pd_ammonia
		}

		# Mirrors vanilla pc_arctic homeworld setup: one +1, one +2, one +3 deposit per category.

		# 3 + 3 energy
		add_deposit = d_pd_ammonia_thermocline
		add_deposit = d_pd_ammonia_lightning_storms
		add_deposit = d_pd_ammonia_polar_ion_columns
		if = {
			limit = { owner = { is_machine_empire = yes } }
			add_deposit = d_pd_ammonia_lightning_storms
		}
		if = {
			limit = { owner = { has_valid_civic = civic_machine_servitor } }
			add_deposit = d_pd_ammonia_thermocline
		}

		# 3 + 3 mining
		add_deposit = d_pd_ammonia_brine_caverns
		add_deposit = d_pd_ammonia_salt_dunes
		add_deposit = d_pd_ammonia_cryolite_reefs

		# 4 + 3 farming (lithoids swap 3 for minerals)
		add_deposit = d_pd_ammonia_organic_haze
		add_deposit = d_pd_ammonia_chemosynthetic_pools
		if = {
			limit = { owner = { is_machine_empire = no } }
			add_deposit = d_pd_ammonia_organic_haze
			if = {
				limit = {
					owner = {
						is_lithoid_or_infernal_empire = yes
					}
				}
				add_deposit = d_pd_ammonia_cryolite_reefs
			}
			else = {
				add_deposit = d_pd_ammonia_ammoniated_kelp_forests
			}
		}
	}

	if = {
		limit = {
			is_planet_class = pc_pd_methane
		}

		# Mirrors the ammonia branch: one +1, one +2, one +3 deposit per category, with machine/servitor energy boosts and a lithoid/infernal farming swap to mining.

		# 3 + 3 energy
		add_deposit = d_pd_methane_tholin_storm_belts
		add_deposit = d_pd_methane_cryovolcanic_vents
		add_deposit = d_pd_methane_supercooled_geysers
		if = {
			limit = { owner = { is_machine_empire = yes } }
			add_deposit = d_pd_methane_cryovolcanic_vents
		}
		if = {
			limit = { owner = { has_valid_civic = civic_machine_servitor } }
			add_deposit = d_pd_methane_tholin_storm_belts
		}

		# 3 + 3 mining
		add_deposit = d_pd_methane_carbonate_caverns
		add_deposit = d_pd_methane_clathrate_shelves
		add_deposit = d_pd_methane_ice_diamonds

		# 4 + 3 farming (lithoids swap 3 for minerals)
		add_deposit = d_pd_methane_tholin_haze
		add_deposit = d_pd_methane_liquid_methane_seas
		if = {
			limit = { owner = { is_machine_empire = no } }
			add_deposit = d_pd_methane_tholin_haze
			if = {
				limit = {
					owner = {
						is_lithoid_or_infernal_empire = yes
					}
				}
				add_deposit = d_pd_methane_ice_diamonds
			}
			else = {
				add_deposit = d_pd_methane_hydrocarbon_marshes
			}
		}
	}
}
