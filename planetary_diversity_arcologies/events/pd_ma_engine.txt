namespace = pdmaengine
#This event sets a flag at game start
event = {
	id = pdmaengine.00
	hide_window = yes
	is_triggered_only = yes
	fire_only_once = yes
	immediate = { set_global_flag = planetarydiversitymorearcologies }
}

planet_event = {
	id = pdmaengine.10
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_pd_capital_deposit = yes
		is_capital = no
	}

	immediate = {
		remove_pd_capital_deposit_effect = yes
	}
}

#this/root = new capital
#from = old capital
planet_event = {
	id = pdmaengine.20
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		exists = from
	}

	immediate = {
		from = {
			if = {
				limit = {
					has_pd_capital_deposit = yes
					is_capital = no
				}
				remove_pd_capital_deposit_effect = yes
			}
		}
	}
}
