triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = {
			is_spiritualist = yes
			is_megacorp = no
			is_death_cult_empire = no
		}
		OR = {
			any_moon = { has_modifier = pd_moon_base_priest }
			AND = {
				is_moon = yes
				solar_system = {
					any_system_planet = {
						has_moon = yes
						any_moon = {
							has_modifier = pd_moon_base_priest
						}
					}
				}
			}
		}
	}
	job_priest_add = $AMOUNT$
	mult = value:num_pd_priest_jobs
}
