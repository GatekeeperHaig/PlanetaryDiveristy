# Each system colony deposit spawned by decision on the main colony, should equal 4 districts on the system colony planet
# So a blocker should also spawn on the system colony to reduce the available districts

# If the system colony has more than 4 available farming districts for example, the Decision to build a farming system colony will be enabled on the main colony
# Same for mining and generator, as long as number of free districts is higher than 4

# Specialized system colonies are also a possibility
# These can be Research, Industrial and Refinery if there are special resource deposits present on the planet

d_system_colony-large = {
	is_for_colonizable = yes
	category = deposit_cat_blockers_natural

	icon = d_system_colony

	drop_weight = {
		weight = 0
	}
	can_be_cleared = {
		always = no
	}
