tech_pd_cryo_toxic_mire = {
	cost = @tier1cost2
	area = society

	weight = @tier1weight2

		deposit_blockers

		}
		modifier = {
			factor = 0

				}
			}
		}
		modifier = {
			factor = 0
			num_owned_planets < 2

	}

	ai_weight = {

			factor = 0

				}

		}

			factor = 0
			num_owned_planets < 2
		}
	}
}
