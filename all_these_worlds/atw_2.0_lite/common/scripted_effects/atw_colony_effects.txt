## Effects to spawn new deposits after finishing the Geographical Survey decision
## Small worlds can spawn between 1 and 2 deposits, and up to 6 on large worlds (but unlikely so many)

add_geo_survey_deposits = {
	# Higher odds of spawning first deposit
	random_list = {
		90 = {
			# spawn first deposit
			spawn_geo_survey_deposit = yes
			set_planet_flag = "first_geo_survey_deposit_spawned"
		}
		10 = {
			# no deposit unless no deposits were spawned at all
			modifier = { factor = 0.00 NOT = { has_planet_flag = "first_geo_survey_deposit_spawned" } }
		}
	}
	# Lower odds of spawning second deposit
	random_list = {
		65 = {
			spawn_geo_survey_deposit = yes
			set_planet_flag = "second_geo_survey_deposit_spawned"
		}
		35 = {
			# no deposit, and increase odds if planet already had a second deposit spawned from a previous iteration
			modifier = { factor = 2.00 has_planet_flag = "first_geo_survey_deposit_spawned" }
			modifier = { factor = 2.00 has_planet_flag = "second_geo_survey_deposit_spawned" }
		}
	}
}
